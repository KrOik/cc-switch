pub fn toggle_app(state: &AppState, server_id: &str, app: AppType, enabled: bool) -> Result<(), AppError> {
    let mut servers = state.db.get_all_mcp_servers()?;

    if let Some(server) = servers.get_mut(server_id) {
        server.apps.set_enabled_for(&app, enabled);
        state.db.save_mcp_server(server)?;

        // ✅ 同步到对应应用
        if enabled {
            Self::sync_server_to_app(state, server, &app)?;
        } else {
            Self::remove_server_from_app(state, server_id, &app)?;
        }
    }

    Ok(())
}

## 6. 关键问题总结

### 高优先级问题（影响核心功能）

**❌ MCP 服务器 CRUD 缺少 live 配置同步**
- 位置：`cc-switch-tui/src/app/actions.rs:246-314`
- 影响：MCP 服务器的添加/编辑/删除/切换不会同步到应用
- 修复：调用 `mcp::sync_single_server_to_*` 和 `mcp::remove_server_from_*`

### 中优先级问题（影响部分功能）

**⚠️ 统一供应商缺少同步逻辑**
- 位置：`cc-switch-tui/src/app/actions.rs:358-418`
- 影响：统一供应商的 CRUD 不会同步到应用
- 修复：需要实现统一供应商的同步逻辑

### 低优先级问题（非关键功能）

**⚠️ Provider 切换缺少 Backfill 机制**
- 影响：用户在应用中的配置修改会丢失
- 修复：切换前保存当前 live 配置到旧 provider

**⚠️ Provider 切换缺少本地 settings 更新**
- 影响：设备级别的 provider 记录不准确
- 修复：调用 `crate::settings::set_current_provider()`

**⚠️ Provider 切换缺少代理接管模式检测**
- 影响：代理接管模式下会多写一次 live 配置
- 修复：检测接管模式，热切换时不写 live 配置
